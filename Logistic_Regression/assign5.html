def cost_function(params, train_data, train_labels):
    '''
    Input: params, model parameters
           train_data, input features
           train_labels, input lables
    '''
    vr=np.vectorize(sigmoid)
    cost = -(np.sum(np.log(vr(np.dot(train_data,params))) * train_labels + np.log(1 - vr(np.dot(train_data,params))) * (1 - train_labels)))/len(train_data)
    
    return cost

def gradient(theta, train_data, train_label):
    vr=np.vectorize(sigmoid)
    gradient = np.dot(train_data.T, (vr(np.dot(train_data,theta)) - train_label)) / len(train_data)
    
    return gradient

def compute_gradient_descent(theta, train_data, train_label, alpha=0.01, niter=1500):
    costs = []
    for i in range(niter):
        theta = theta - (alpha * gradient(theta, train_data, train_label))
        costs.append(cost_function(theta, train_data, train_label))   
    new_theta = theta
    return new_theta, costs

def cost_function(theta, X, y, reg_param=250):
    H = np.dot(X, theta.reshape(6,1))   
    itheta = theta[1:].reshape((1,5))  
    cost = np.sum(np.dot((H - y).T, (H - y)) + reg_param * np.dot(itheta, itheta.T)) / (2 * len(X))
    grad_zero = np.dot(X[:,0].T, (H - y)) / len(X)
    grad_notzero = (np.dot(X[:,1:].T, (H - y)) + reg_param * itheta.T) / len(X)
    gradient = np.concatenate((grad_zero, grad_notzero.sum(axis = 1)), axis = 0)
    return cost, gradient.flatten()
